bool canJump(vector<int>& nums) {

        int n = nums.size();

        bool s[n];

        for(int i=0;i<n;i++)
        {
            s[i] = false;
        }


        s[0] = true;

        for(int i=0;i<n-1;i++)
        {
            if(s[i] == true)
            {
                for(int j = i+1; j<=(i+nums[i]) && j<n  ; j++)
                {
                    s[j] = true;
                }
            }
        }

        return s[n-1];
        
    }
